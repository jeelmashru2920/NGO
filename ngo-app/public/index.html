SQL Query — Top Donors
SELECT d.name, d.city, COUNT(don.id) AS donations,
  SUM(don.amount) AS total
FROM donors d
JOIN donations don ON d.id = don.donor_id
GROUP BY d.id ORDER BY total DESC LIMIT 5;
SQL Query — Activity Participation
SELECT a.category, COUNT(*) AS count
FROM activities a
GROUP BY a.category
ORDER BY count DESC;
